/** * PAINEL OPERACIONAL — BACKEND GOOGLE SHEETS + AGENDA + LOGIN * --------------------------------------------------- * Como instalar: * 1. Crie uma Google Planilha nova (sheets.new). * 2. Menu Extensões > Apps Script. * 3. Apague o conteúdo do arquivo Code.gs e cole TODO o conteúdo deste arquivo. * 4. No menu suspenso ao lado de "Executar" (topo do editor), selecione * a função "autorizarAgenda" e clique em "Executar". * 5. Aceite a tela de permissão ("App não verificado" > Configurações * avançadas > Acessar [nome do projeto] (não seguro) > Permitir). * Isso concede acesso à sua Agenda do Google. * 6. Selecione a função "criarPrimeiroAdmin" no mesmo menu suspenso, * edite antes o e-mail/senha dentro dela (mais abaixo neste arquivo) * e clique em "Executar" para criar seu usuário administrador. * 7. Clique em "Implantar" > "Nova implantação". * 8. Tipo: "App da Web". * 9. Executar como: "Eu" (sua conta). * 10. Quem pode acessar: "Qualquer pessoa". * 11. Clique em Implantar, autorize novamente se pedido. * 12. Copie a URL gerada (termina em /exec) e cole no dashboard * HTML, no botão "Configurar Google Sheets". * * Toda atividade do Painel Geral que tiver um Prazo preenchido vira um * evento na sua Google Agenda automaticamente, atualizado a cada * alteração de status/prazo — sem duplicar eventos. Se a atividade * tiver e-mails no campo "convidados" (preenchido no dashboard), esses * e-mails são convidados no evento com permissão de edição (mas nunca * de exclusão — isso é o padrão do Google Calendar). * * Os registros de Criação de Roedores e Treinamento de Aves também * viram eventos na Agenda automaticamente, um para cada linha com data * preenchida (nascimento, abate, entrada, saída, treinamento, feedback). * * LOGIN E PERMISSÕES: * - Ninguém acessa dado nenhum sem fazer login (e-mail + senha). * - Papel "adm": edita tudo, exclui registros e pode restaurar os * dados originais. * - Qualquer outro papel (ex: "editor"): pode ver e editar/adicionar * registros em todos os módulos, mas NÃO pode excluir registros nem * restaurar os dados — essas duas ações são bloqueadas no servidor, * não só escondidas na tela. * - Sessões expiram sozinhas depois de 6 horas (é preciso logar de novo). * - Para criar mais usuários, use o painel "Gerenciar usuários" dentro * do próprio dashboard (só aparece para administradores) — não * precisa mais editar código para cada pessoa nova. A função * "criarNovoUsuario" mais abaixo continua funcionando como alternativa * manual, caso prefira. * * Cada vez que você editar este código, é preciso gerar uma NOVA * implantação (ou editar a implantação existente em "Gerenciar * implantações") para as mudanças valerem — só salvar o arquivo * não atualiza a URL já publicada. * * Para testar rapidamente se está no ar: cole a URL /exec direto * na barra de endereço do navegador. Deve aparecer algo como * {"error":"chave inválida: undefined"} — isso é o esperado e * significa que o servidor está respondendo corretamente. */ const SHEET_MAP = { 'painel-atividades': 'Atividades', 'log-roedores-nascimento': 'Roedores_Nascimento', 'log-roedores-abate': 'Roedores_Abate', 'log-roedores-entrada': 'Roedores_Entrada', 'log-roedores-saida': 'Roedores_Saida', 'log-aves-treinamento': 'Aves_Treinamento', 'log-aves-feedback': 'Aves_Feedback' }; const USERS_SHEET = 'Usuarios'; const ADMIN_ROLE = 'adm'; const SESSION_TTL_SECONDS = 6 * 60 * 60; // 6 horas function doGet(e) { try { if (e.parameter.action === 'ping') { return jsonOutput({ ok: true }); } const session = getSession_(e.parameter.token); if (!session) return jsonOutput({ error: 'unauthenticated' }); const key = e.parameter.key; const sheetName = SHEET_MAP[key]; if (!sheetName) return jsonOutput({ error: 'chave inválida: ' + key }); const rows = readSheet(sheetName); if (rows === null) { return jsonOutput({ value: null }); } return jsonOutput({ value: JSON.stringify(rows) }); } catch (err) { return jsonOutput({ error: String(err) }); } } function doPost(e) { try { const body = JSON.parse(e.postData.contents); if (body.action === 'login') { return handleLogin_(body); } const session = getSession_(body.token); if (!session) { return jsonOutput({ error: 'unauthenticated' }); } if (body.action === 'reset') { if (session.papel !== ADMIN_ROLE) { return jsonOutput({ error: 'Somente administradores podem restaurar os dados.' }); } return handleReset_(body); } if (body.action === 'createUser') { if (session.papel !== ADMIN_ROLE) { return jsonOutput({ error: 'Somente administradores podem criar usuários.' }); } return handleCreateUser_(body); } if (body.action === 'listUsers') { if (session.papel !== ADMIN_ROLE) { return jsonOutput({ error: 'Somente administradores podem ver a lista de usuários.' }); } return handleListUsers_(); } if (body.action === 'deleteUser') { if (session.papel !== ADMIN_ROLE) { return jsonOutput({ error: 'Somente administradores podem remover usuários.' }); } return handleDeleteUser_(body, session); } const key = body.key; const sheetName = SHEET_MAP[key]; if (!sheetName) return jsonOutput({ error: 'chave inválida: ' + key }); const newArr = JSON.parse(body.value); if (session.papel !== ADMIN_ROLE) { const currentArr = readSheet(sheetName) || []; const currentIds = currentArr.map(function (r) { return String(r.id); }); const newIds = newArr.map(function (r) { return String(r.id); }); const removed = currentIds.filter(function (id) { return newIds.indexOf(id) === -1; }); if (removed.length > 0) { return jsonOutput({ error: 'Somente administradores podem excluir registros.' }); } } writeSheet(sheetName, newArr); let calendarWarning = null; try { if (key === 'painel-atividades') { syncActivitiesToCalendar_(newArr); } else if (CALENDAR_LOG_CONFIG[key]) { syncLogToCalendar_(key, newArr); } } catch (calErr) { // Nunca deixa um problema na Agenda impedir o salvamento na planilha. calendarWarning = String(calErr); } return jsonOutput({ ok: true, calendarWarning: calendarWarning }); } catch (err) { return jsonOutput({ error: String(err) }); } } /** * Cria/atualiza um evento no Google Agenda (da conta usada para implantar * o script) para cada atividade do Painel Geral que tenha um prazo * preenchido. Usa uma "tag" no evento (painelId) para reconhecer eventos * já criados e atualizá-los, em vez de duplicar a cada salvamento. * * Permissões: * - Quem cria o evento (a conta do script) já é o organizador, com * permissão total — isso é automático no Google Calendar. * - Os convidados (campo "emails" da atividade, separados por vírgula) * recebem permissão de EDITAR o evento (setGuestsCanModify), mas o * Google Calendar nunca permite que um convidado exclua o evento do * calendário do organizador — eles só conseguem removê-lo da própria * agenda. Ou seja, "editar sem excluir dados" já é o comportamento * padrão do Google, não precisa ser simulado. * * ATENÇÃO — autorização única necessária: * Esta função usa o serviço CalendarApp, que precisa de uma permissão * que ainda não foi concedida ao script. Antes de usar, no editor do * Apps Script: selecione a função "autorizarAgenda" no menu suspenso ao * lado do botão "Executar", clique em Executar e aceite a tela de * permissão (Configurações avançadas > Acessar [projeto] (não seguro) > * Permitir). Depois disso, gere uma nova implantação (Gerenciar * implantações > editar > Nova versão) para o app da web passar a ter * essa permissão também. */ function syncActivitiesToCalendar_(items) { const cal = CalendarApp.getDefaultCalendar(); items.forEach(function (item) { if (!item.prazo) return; const id = String(item.id); const title = '[Operacional] ' + item.title; const date = new Date(item.prazo + 'T09:00:00'); if (isNaN(date.getTime())) return; const end = new Date(date.getTime() + 30 * 60000); const description = 'Área: ' + item.area + '\nStatus: ' + item.status + '\nPróxima ação: ' + item.next + '\nResponsável: ' + item.responsavel; const guestEmails = parseEmails_(item.emails); let ev = findEventByTag_(cal, 'painelId', id); if (ev) { ev.setTitle(title); ev.setTime(date, end); ev.setDescription(description); } else { ev = cal.createEvent(title, date, end, { description: description }); ev.setTag('painelId', id); } syncGuests_(ev, guestEmails); ev.setGuestsCanModify(true); ev.setGuestsCanInviteOthers(false); ev.setGuestsCanSeeGuests(true); }); } function parseEmails_(str) { if (!str) return []; return String(str).split(/[,;]/) .map(function (s) { return s.trim(); }) .filter(function (s) { return /^[^@\s]+@[^@\s]+\.[^@\s]+$/.test(s); }); } function syncGuests_(ev, emails) { const current = ev.getGuestList().map(function (g) { return g.getEmail(); }); emails.forEach(function (email) { if (current.indexOf(email) === -1) { ev.addGuest(email); } }); } function findEventByTag_(cal, tagKey, tagValue) { const now = new Date(); const start = new Date(now.getFullYear() - 1, 0, 1); const end = new Date(now.getFullYear() + 2, 0, 1); const events = cal.getEvents(start, end); for (var i = 0; i < events.length; i++) { if (events[i].getTag(tagKey) === tagValue) return events[i]; } return null; } /** * Configuração de como cada módulo (Roedores / Aves) vira evento na * Agenda: título e descrição são montados a partir dos campos de cada * registro. Cada registro precisa ter um campo "data" preenchido — * registros sem data são ignorados. */ const CALENDAR_LOG_CONFIG = { 'log-roedores-nascimento': { build: function (item) { return { title: '[Roedores] Nascimento — ' + (item.linhagem || 'sem linhagem'), description: 'Quantidade: ' + (item.quantidade || '') + '\nLinhagem/lote: ' + (item.linhagem || '') + '\nObservação: ' + (item.obs || '') }; } }, 'log-roedores-abate': { build: function (item) { return { title: '[Roedores] Abate — ' + (item.linhagem || 'sem linhagem'), description: 'Quantidade: ' + (item.quantidade || '') + '\nLinhagem/lote: ' + (item.linhagem || '') + '\nMotivo/destino: ' + (item.motivo || '') }; } }, 'log-roedores-entrada': { build: function (item) { return { title: '[Roedores] Entrada — ' + (item.linhagem || 'sem linhagem'), description: 'Quantidade: ' + (item.quantidade || '') + '\nLinhagem/lote: ' + (item.linhagem || '') + '\nOrigem/fornecedor: ' + (item.origem || '') }; } }, 'log-roedores-saida': { build: function (item) { return { title: '[Roedores] Saída — ' + (item.linhagem || 'sem linhagem'), description: 'Quantidade: ' + (item.quantidade || '') + '\nLinhagem/lote: ' + (item.linhagem || '') + '\nDestino: ' + (item.destino || '') }; } }, 'log-aves-treinamento': { build: function (item) { return { title: '[Aves] Treinamento — ' + (item.ave_id || 'sem identificação'), description: 'Peso inicial: ' + (item.peso_inicial || '') + ' g' + '\nPeso final: ' + (item.peso_final || '') + ' g' + '\nAlimentação: ' + (item.alimentacao || '') + '\nObservação: ' + (item.obs || '') }; } }, 'log-aves-feedback': { build: function (item) { return { title: '[Aves] Feedback — ' + (item.ave_id || 'geral'), description: 'Responsável: ' + (item.responsavel || '') + '\nFeedback: ' + (item.feedback || '') }; } } }; /** * Cria/atualiza um evento na Agenda para cada registro de Roedores/Aves * que tenha uma data preenchida. Mesma lógica de "tag" da função de * atividades: reconhece e atualiza o evento em vez de duplicar. */ function syncLogToCalendar_(key, items) { const cfg = CALENDAR_LOG_CONFIG[key]; if (!cfg) return; const cal = CalendarApp.getDefaultCalendar(); items.forEach(function (item) { if (!item.data) return; const date = new Date(item.data + 'T09:00:00'); if (isNaN(date.getTime())) return; const end = new Date(date.getTime() + 30 * 60000); const tagValue = key + ':' + String(item.id); const built = cfg.build(item); let ev = findEventByTag_(cal, 'logTag', tagValue); if (ev) { ev.setTitle(built.title); ev.setTime(date, end); ev.setDescription(built.description); } else { ev = cal.createEvent(built.title, date, end, { description: built.description }); ev.setTag('logTag', tagValue); } }); } /** * Rode esta função manualmente UMA VEZ pelo editor do Apps Script * (menu suspenso > autorizarAgenda > Executar) só para disparar a tela * de permissão da Agenda. Ela não faz nada além de acessar a agenda * padrão — é só para conceder a permissão de forma segura e visível. */ function autorizarAgenda() { const cal = CalendarApp.getDefaultCalendar(); Logger.log('Agenda autorizada: ' + cal.getName()); } function getSheet_(name, createIfMissing) { const ss = SpreadsheetApp.getActiveSpreadsheet(); let sheet = ss.getSheetByName(name); if (!sheet && createIfMissing) { sheet = ss.insertSheet(name); } return sheet || null; } function readSheet(name) { const sheet = getSheet_(name, false); if (!sheet) return null; const values = sheet.getDataRange().getValues(); if (values.length < 2) return []; const headers = values[0]; return values.slice(1) .map(function (row) { const obj = {}; headers.forEach(function (h, i) { let v = row[i]; if (typeof v === 'string' && v !== '' && !isNaN(v) && !/^0[0-9]/.test(v)) { v = Number(v); } obj[h] = v; }); return obj; }) .filter(function (obj) { return Object.keys(obj).some(function (k) { return obj[k] !== '' && obj[k] !== null; }); }); } function writeSheet(name, arr) { const sheet = getSheet_(name, true); sheet.clear(); if (!arr || arr.length === 0) return; const headerSet = {}; arr.forEach(function (o) { Object.keys(o).forEach(function (k) { headerSet[k] = true; }); }); const headers = Object.keys(headerSet); const rows = arr.map(function (o) { return headers.map(function (h) { return (o[h] !== undefined && o[h] !== null) ? o[h] : ''; }); }); const totalRows = rows.length + 1; sheet.getRange(1, 1, totalRows, headers.length).setNumberFormat('@'); sheet.getRange(1, 1, 1, headers.length).setValues([headers]); if (rows.length) { sheet.getRange(2, 1, rows.length, headers.length).setValues(rows); } } /** * Restaura os dados originais (semente) da aba Atividades e limpa todas * as abas de Roedores/Aves. Só é chamada quando doPost já confirmou que * a sessão pertence a um usuário com papel "adm". */ function handleReset_(body) { Object.keys(SHEET_MAP).forEach(function (key) { const sheetName = SHEET_MAP[key]; if (key === 'painel-atividades' && body.seed) { writeSheet(sheetName, JSON.parse(body.seed)); } else { const sheet = getSheet_(sheetName, false); if (sheet) sheet.clear(); } }); return jsonOutput({ ok: true }); } /* ===================================================================== AUTENTICAÇÃO — usuários, senhas (com salt), sessões e login ===================================================================== */ /** * Calcula um hash SHA-256 de "salt:senha", em hexadecimal. Não é o * mesmo nível de proteção de bcrypt/scrypt, mas evita guardar senhas em * texto puro na planilha — suficiente para o uso interno deste painel. */ function hashPassword_(password, salt) { const bytes = Utilities.computeDigest(Utilities.DigestAlgorithm.SHA_256, salt + ':' + password); return bytes.map(function (b) { const v = (b < 0 ? b + 256 : b).toString(16); return v.length === 1 ? '0' + v : v; }).join(''); } function generateSalt_() { return Utilities.getUuid(); } function generateToken_() { return Utilities.getUuid().replace(/-/g, '') + Utilities.getUuid().replace(/-/g, ''); } function getUsersSheet_() { return getSheet_(USERS_SHEET, true); } /** Procura um usuário pelo e-mail (case-insensitive). Retorna null se não achar. */ function findUserByEmail_(email) { const sheet = getUsersSheet_(); const values = sheet.getDataRange().getValues(); if (values.length < 2) return null; const headers = values[0]; const emailIdx = headers.indexOf('email'); if (emailIdx === -1) return null; for (var i = 1; i < values.length; i++) { if (String(values[i][emailIdx]).toLowerCase() === String(email).toLowerCase()) { const obj = {}; headers.forEach(function (h, idx) { obj[h] = values[i][idx]; }); obj._row = i + 1; return obj; } } return null; } /** * Cria (ou atualiza, se o e-mail já existir) um usuário. Chamada pelas * funções "criarPrimeiroAdmin" e "criarNovoUsuario" mais abaixo — não é * exposta via doGet/doPost, só pode ser rodada manualmente pelo editor * do Apps Script. */ function criarUsuario_(email, password, nome, papel) { const sheet = getUsersSheet_(); if (sheet.getLastRow() === 0) { sheet.getRange(1, 1, 1, 5).setValues([['email', 'nome', 'papel', 'salt', 'hash']]); sheet.getRange(1, 1, sheet.getMaxRows(), 5).setNumberFormat('@'); } const salt = generateSalt_(); const hash = hashPassword_(password, salt); const existing = findUserByEmail_(email); if (existing) { sheet.getRange(existing._row, 1, 1, 5).setValues([[email, nome, papel, salt, hash]]); Logger.log('Usuário atualizado: ' + email + ' (papel: ' + papel + ')'); } else { sheet.appendRow([email, nome, papel, salt, hash]); Logger.log('Usuário criado: ' + email + ' (papel: ' + papel + ')'); } } /** * PASSO A PASSO: edite o e-mail e a senha abaixo para os que você vai * usar, salve (Ctrl+S), selecione esta função no menu suspenso ao lado * de "Executar" e clique em Executar. Depois de rodar uma vez, pode * apagar a senha em texto puro daqui se quiser (o usuário já foi * salvo com hash na planilha). */ function criarPrimeiroAdmin() { criarUsuario_('seuemail@exemplo.com', 'TROQUE_ESTA_SENHA', 'Duílio', ADMIN_ROLE); } /** * Use esta função como modelo para criar os demais usuários da equipe. * Edite os quatro valores, rode pelo editor (mesmo processo do * "criarPrimeiroAdmin"), e repita para cada pessoa nova — troque os * dados a cada execução. * * Papel "adm" = acesso total (inclusive excluir e restaurar dados). * Qualquer outro texto no papel (ex: "editor", "coordenador", * "colaborador") = pode ver e editar/adicionar, mas nunca excluir nem * restaurar — essa regra é fixa e verificada no servidor. */ function criarNovoUsuario() { criarUsuario_('pessoa@exemplo.com', 'SenhaTemporaria123', 'Nome da Pessoa', 'editor'); } /** * Cria um usuário a partir do painel "Gerenciar usuários" do dashboard * (em vez de precisar editar código no Apps Script). Só é chamada * depois que doPost já confirmou que quem pediu é administrador. */ function handleCreateUser_(body) { const email = String(body.email || '').trim(); const password = String(body.password || ''); const nome = String(body.nome || '').trim(); const papel = String(body.papel || '').trim() || 'editor'; if (!email || !password || !nome) { return jsonOutput({ error: 'Preencha nome, e-mail e senha.' }); } if (password.length < 6) { return jsonOutput({ error: 'A senha precisa ter pelo menos 6 caracteres.' }); } if (findUserByEmail_(email)) { return jsonOutput({ error: 'Já existe um usuário com esse e-mail.' }); } criarUsuario_(email, password, nome, papel); return jsonOutput({ ok: true }); } /** Lista os usuários cadastrados (sem salt/hash) para o painel de administração. */ function handleListUsers_() { const sheet = getUsersSheet_(); const values = sheet.getDataRange().getValues(); if (values.length < 2) return jsonOutput({ ok: true, users: [] }); const headers = values[0]; const emailIdx = headers.indexOf('email'); const nomeIdx = headers.indexOf('nome'); const papelIdx = headers.indexOf('papel'); const users = []; for (var i = 1; i < values.length; i++) { users.push({ email: values[i][emailIdx], nome: values[i][nomeIdx], papel: values[i][papelIdx] }); } return jsonOutput({ ok: true, users: users }); } /** * Remove um usuário pelo e-mail. Nunca deixa remover a própria conta * logada, nem remover o último administrador restante (pra evitar * ficar todo mundo trancado para fora do painel). */ function handleDeleteUser_(body, session) { const email = String(body.email || '').trim(); if (!email) return jsonOutput({ error: 'E-mail não informado.' }); if (email.toLowerCase() === String(session.email).toLowerCase()) { return jsonOutput({ error: 'Você não pode remover seu próprio usuário.' }); } const existing = findUserByEmail_(email); if (!existing) return jsonOutput({ error: 'Usuário não encontrado.' }); if (String(existing.papel) === ADMIN_ROLE) { const sheet = getUsersSheet_(); const values = sheet.getDataRange().getValues(); const headers = values[0]; const papelIdx = headers.indexOf('papel'); var adminCount = 0; for (var i = 1; i < values.length; i++) { if (String(values[i][papelIdx]) === ADMIN_ROLE) adminCount++; } if (adminCount <= 1) { return jsonOutput({ error: 'Não é possível remover o único administrador.' }); } } getUsersSheet_().deleteRow(existing._row); return jsonOutput({ ok: true }); } /** Login: valida e-mail/senha e devolve um token de sessão válido por 6h. */ function handleLogin_(body) { const email = String(body.email || '').trim(); const password = String(body.password || ''); const user = findUserByEmail_(email); if (!user) { return jsonOutput({ error: 'E-mail ou senha inválidos.' }); } const hash = hashPassword_(password, user.salt); if (hash !== user.hash) { return jsonOutput({ error: 'E-mail ou senha inválidos.' }); } const token = generateToken_(); const session = { email: user.email, nome: user.nome, papel: user.papel }; CacheService.getScriptCache().put('session_' + token, JSON.stringify(session), SESSION_TTL_SECONDS); return jsonOutput({ ok: true, token: token, nome: user.nome, papel: user.papel }); } /** Recupera a sessão associada a um token, ou null se inválida/expirada. */ function getSession_(token) { if (!token) return null; const raw = CacheService.getScriptCache().get('session_' + token); if (!raw) return null; try { return JSON.parse(raw); } catch (e) { return null; } } function jsonOutput(obj) { return ContentService .createTextOutput(JSON.stringify(obj)) .setMimeType(ContentService.MimeType.JSON);
